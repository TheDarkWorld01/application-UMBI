import 'package:flutter/material.dart';
import 'package:umbiapps/class/shopping_cart.dart';

class CartItemSamples extends StatelessWidget {
   final ValueNotifier<int>? quantity;

  CartItemSamples({Key? key, this.quantity}) : super(key: key);

  void _increaseQuantity() {
    if (quantity != null) {
      quantity!.value++;
    }
  }

  void _decreaseQuantity() {
    if (quantity != null && quantity!.value > 1) {
      quantity!.value--;
    }
  }

  void _removeItem() {
    if (quantity != null) {
      // Menghapus item dari keranjang dengan menggunakan indeks
      final int itemIndex = getItemIdToRemove(); // Anda perlu mengganti ini dengan logika Anda sendiri untuk mendapatkan indeks item yang akan dihapus.
      if (itemIndex >= 0) {
        quantity!.value--;
        // Menghapus item dari daftar keranjang dengan indeks
        ShoppingCart.cartItems.removeAt(itemIndex);
      }
    }
  }
   int getItemIdToRemove() {
    // Misalnya, jika Anda memiliki atribut id di CartItemSamples, Anda dapat mengembalikan nilainya.
    // return id;
    return -1; // Default return -1 jika tidak ada item yang harus dihapus.
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Container(
          height: 110,
          margin: EdgeInsets.symmetric(horizontal: 15, vertical: 10),
          padding: EdgeInsets.all(10),
          decoration: BoxDecoration(
            color: Colors.white,
            borderRadius: BorderRadius.circular(10),
          ),
          child: Row(
            children: [
              Radio(
                value: "",
                groupValue: "",
                activeColor: Color(0xFF4C53A5),
                onChanged: (index) {},
              ),
              Container(
                height: 70,
                width: 70,
                margin: EdgeInsets.only(right: 15),
                child: Image.asset("asset/images/1.png"),
              ),
              Padding(
                padding: EdgeInsets.symmetric(vertical: 10),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    Text(
                      "Fried Chicken",
                      style: TextStyle(
                        fontSize: 18,
                        fontWeight: FontWeight.bold,
                        color: Color(0xFF4C53A5),
                      ),
                    ),
                    Text(
                      "\Rp.18.000",
                      style: TextStyle(
                        fontSize: 16,
                        fontWeight: FontWeight.bold,
                        color: Color(0xFF4C53A5),
                      ),
                    )
                  ],
                ),
              ),
              Spacer(),
              Padding(
                padding: EdgeInsets.symmetric(vertical: 5),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.end,
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    GestureDetector(
                      onTap: _removeItem,
                      child: Icon(
                        Icons.delete,
                        color: Colors.red,
                      ),
                    ),
                    Row(
                      children: [
                        Container(
                          padding: EdgeInsets.all(4),
                          decoration: BoxDecoration(
                            color: Colors.white,
                            borderRadius: BorderRadius.circular(20),
                            boxShadow: [
                              BoxShadow(
                                color: Colors.grey.withOpacity(0.5),
                                spreadRadius: 1,
                                blurRadius: 10,
                              ),
                            ],
                          ),
                          child: GestureDetector(
                            onTap: _increaseQuantity,
                            child: Icon(
                              Icons.add,
                              size: 18,
                            ),
                          ),
                        ),
                        Container(
                          margin: EdgeInsets.symmetric(horizontal: 10),
                          child: ValueListenableBuilder<int>(
                            valueListenable: quantity ??
                                ValueNotifier<int>(
                                    0), // Tambahkan pengecekan kondisi
                            builder: (context, value, child) {
                              return Text(
                                value.toString().padLeft(2, '0'),
                                style: TextStyle(
                                  fontSize: 15,
                                  fontWeight: FontWeight.bold,
                                  color: Color(0xFF4C53A5),
                                ),
                              );
                            },
                          ),
                        ),
                        Container(
                          padding: EdgeInsets.all(4),
                          decoration: BoxDecoration(
                            color: Colors.white,
                            borderRadius: BorderRadius.circular(20),
                            boxShadow: [
                              BoxShadow(
                                color: Colors.grey.withOpacity(0.5),
                                spreadRadius: 1,
                                blurRadius: 10,
                              ),
                            ],
                          ),
                          child: GestureDetector(
                            onTap: _decreaseQuantity,
                            child: Icon(
                              Icons.remove,
                              size: 18,
                            ),
                          ),
                        ),
                      ],
                    ),
                  ],
                ),
              ),
            ],
          ),
        ),
      ],
    );
  }
}
